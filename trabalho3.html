           Aritméticos
           + - * / %  // + maior; - menor; * multiplicação ; / divisão ; % resto da divisão
           Relacionais
           > < >= <= == !=  // > maior que; < menor que; >= maior que ou igual a; < menor que ou igual a; == igual; != diferente
           Lógicos 
           && || !  //&& -> E; || -> OU; ! -> FALSO;
